```python
import numpy as np

class DataGenerator:
    def generate(self, distribution_type, size, params=None):
        """Генерация данных указанного распределения"""
        if params is None:
            params = {}
            
        if distribution_type == "normal":
            return np.random.normal(
                loc=params.get('mean', 0),
                scale=params.get('std', 1),
                size=size
            )
        elif distribution_type == "uniform":
            return np.random.uniform(
                low=params.get('low', 0),
                high=params.get('high', 1),
                size=size
            )
        elif distribution_type == "gamma":
            return np.random.gamma(
                shape=params.get('shape', 2),
                scale=params.get('scale', 1),
                size=size
            )
        else:
            raise ValueError(f"Неподдерживаемый тип распределения: {distribution_type}")
```
