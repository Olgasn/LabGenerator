import threading

class Warehouse:
    def __init__(self, capacity=100):
        self.capacity = capacity
        self.storage = {}
        self.item_count = 0
        self.lock = threading.Lock()

    def add_item(self, item_id, name, quantity):
        if not isinstance(quantity, int) or quantity <= 0:
            raise ValueError("Количество должно быть положительным целым числом")
        with self.lock:
            if self.item_count >= self.capacity:
                raise ValueError("Склад переполнен")
            if item_id in self.storage:
                self.storage[item_id]['quantity'] += quantity
            else:
                self.storage[item_id] = {'name': name, 'quantity': quantity}
                self.item_count += 1

    def remove_item(self, item_id, quantity):
        if not isinstance(quantity, int) or quantity <= 0:
            raise ValueError("Количество должно быть положительным целым числом")
        with self.lock:
            if item_id not in self.storage:
                raise KeyError("Товар не найден")
            if self.storage[item_id]['quantity'] < quantity:
                raise ValueError("Недостаточно товара на складе")
            self.storage[item_id]['quantity'] -= quantity
            if self.storage[item_id]['quantity'] == 0:
                del self.storage[item_id]
                self.item_count -= 1

    def get_item_info(self, item_id):
        with self.lock:
            return self.storage.get(item_id, None).copy() if item_id in self.storage else None

class SpecialWarehouse(Warehouse):

        if not 0 <= discount_percent <= 100:

        with self.lock:
            if item_id in self.storage:
                self.storage[item_id]['discount'] = discount_percent

   - `prioritizeRisks(risks: List<Risk>): List<Risk>` - приоритезация рисков
   - `addRule(rule: RiskRule): void` - добавление нового правила анализа
   - `getRiskHistory(): List<RiskEvent>` - получение истории рисков

2. **Класс RiskMitigator** - обработчик мер защиты
   - `applyMitigation(risk: Risk): MitigationResult` - применение мер защиты
   - `rollbackAction(action: Action): void` - откат ошибочных операций
   - `getActionQueue(): PriorityQueue<Action>` - получение очереди действий
   - `registerStrategy(strategy: MitigationStrategy): void` - регистрация стратегии

3. **Класс RiskMonitor** - система мониторинга
   - `collectMetrics(): List<Metric>` - сбор метрик системы
   - `detectAnomalies(metrics: List<Metric>): List<Anomaly>` - обнаружение аномалий
   - `subscribe(listener: EventListener): void` - подписка на события
   - `getSystemStatus(): SystemStatus` - получение статуса системы
