```python
import logging

class HashTable:
    def __init__(self, size=10):
        self.size = size
        self.table = [[] for _ in range(size)]
        self.logger = logging.getLogger('HashTable')
        self.logger.setLevel(logging.DEBUG)
        
        # Настройка обработчика для вывода в консоль
        handler = logging.StreamHandler()
        formatter = logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
        handler.setFormatter(formatter)
        self.logger.addHandler(handler)
    
    def _hash(self, key):
        hash_value = hash(key) % self.size
        self.logger.debug(f"Вычислен хеш для ключа {key}: {hash_value}")
        return hash_value
    
    def add_item(self, product_id, product):
        self.logger.info(f"Добавление товара: ID={product_id}")
        index = self._hash(product_id)
        
        for i, item in enumerate(self.table[index]):
            if item[0] == product_id:
                self.logger.warning(f"Обновление существующего товара: ID={product_id}")
                self.table[index][i] = (product_id, product)
                return
        
        if len(self.table[index]) >= 10:  # Ограничение на длину цепочки
            self.logger.error("Достигнуто максимальное количество коллизий для индекса {index}")
            raise OverflowError("Превышена максимальная длина цепочки")
            
        self.table[index].append((product_id, product))
        self.logger.debug(f"Товар добавлен: ID={product_id}, индекс={index}")
    
    def get_item(self, product_id):
        self.logger.info(f"Поиск товара: ID={product_id}")
        index = self._hash(product_id)
        
        for item in self.table[index]:
            if item[0] == product_id:
                self.logger.debug(f"Товар найден: ID={product_id}")
                return item[1]
                
        self.logger.warning(f"Товар не найден: ID={product_id}")
        return None
    
    def remove_item(self, product_id):
        self.logger.info(f"Удаление товара: ID={product_id}")
        index = self._hash(product_id)
        
        for i, item in enumerate(self.table[index]):
            if item[0] == product_id:
                removed = self.table[index].pop(i)
                self.logger.debug(f"Товар удален: ID={product_id}")
                return removed
                
        self.logger.warning(f"Товар для удаления не найден: ID={product_id}")
        return None
