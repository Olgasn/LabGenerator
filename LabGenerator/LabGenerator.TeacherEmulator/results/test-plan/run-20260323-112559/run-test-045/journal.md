public class BinarySearch {
    public int search(int[] array, int target) {
        if (array == null) return -1;
        
        int left = 0;
        int right = array.length - 1;
        
        while (left <= right) {
            int mid = (left + right) / 2;
            
            if (array[mid] == target) {
                return mid;
            } else if (array[mid] < target) {
