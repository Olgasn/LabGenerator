        if (book == null) {
            throw new IllegalArgumentException("Book cannot be null");
        }
        if (books.containsKey(book.getIsbn())) {
            throw new IllegalStateException("Book with ISBN " + book.getIsbn() + " already exists");
        }
        books.put(book.getIsbn(), book);
    }
    
    public boolean borrowBook(String isbn, String userId) {
        if (isbn == null || userId == null) {
            throw new IllegalArgumentException("ISBN and User ID cannot be null");
        }
        if (!users.containsKey(userId)) {
            throw new IllegalArgumentException("User not found: " + userId);
        }
        Book book = books.get(isbn);
        if (book != null && book.isAvailable()) {
            book.setAvailable(false);
            book.setBorrowedBy(userId);
            
            // Добавление записи в историю
            BorrowRecord record = new BorrowRecord(isbn, userId, LocalDate.now());
            borrowHistory.computeIfAbsent(userId, k -> new ArrayList<>()).add(record);

        }

        if (isbn == null) {

    }
