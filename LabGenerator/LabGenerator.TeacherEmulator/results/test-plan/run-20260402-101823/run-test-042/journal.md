import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.stream.Collectors;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class ApplianceManager {
    private static final Logger logger = LoggerFactory.getLogger(ApplianceManager.class);
    private Map<String, Appliance> appliances = new HashMap<>();
    
    public void addAppliance(String id, String type, int powerConsumption) {
        logger.info("Добавление прибора: id={}, type={}, power={}", id, type, powerConsumption);
        if (powerConsumption <= 0) {
            String errorMsg = "Потребляемая мощность должна быть положительной";
            logger.error(errorMsg);
            throw new IllegalArgumentException(errorMsg);
        }
        if (appliances.containsKey(id)) {
            String errorMsg = "Прибор с id=" + id + " уже существует";
            logger.error(errorMsg);
            throw new IllegalArgumentException(errorMsg);
        }
        appliances.put(id, new Appliance(id, type, powerConsumption));
        logger.debug("Прибор успешно добавлен: {}", id);
    }
    
    public void removeAppliance(String id) {
        logger.info("Удаление прибора: id={}", id);
        if (id == null) {
            logger.warn("Попытка удаления прибора с null id");
            return;
        }
        Appliance removed = appliances.remove(id);
        if (removed != null) {
            logger.debug("Прибор успешно удален: {}", id);
        } else {
            logger.warn("Прибор с id={} не найден", id);
        }
    }
    
    public int getTotalPowerConsumption() {
        logger.info("Расчет общего энергопотребления");
        int total = appliances.values().stream()
                .filter(Appliance::isActive)
                .mapToInt(Appliance::getPowerConsumption)
                .sum();
        logger.debug("Общее энергопотребление: {}", total);
        return total;
    }
    
    public List<Appliance> getAppliancesByType(String type) {
        logger.info("Поиск приборов по типу: {}", type);
        if (type == null) {
            logger.warn("Поиск по null типу прибора");
            return List.of();
        }
        return appliances.values().stream()
                .filter(a -> type.equalsIgnoreCase(a.getType()))
                .collect(Collectors.toList());
    }
    
    public void toggleAppliance(String id) {
        logger.info("Переключение состояния прибора: id={}", id);
        if (id == null) {
            logger.warn("Попытка переключения прибора с null id");
            return;
        }
        Appliance appliance = appliances.get(id);
        if (appliance != null) {
            appliance.setActive(!appliance.isActive());
            logger.debug("Состояние прибора {} изменено на {}", id, appliance.isActive() ? "вкл" : "выкл");
        } else {
            logger.warn("Прибор с id={} не найден", id);
        }
    }
